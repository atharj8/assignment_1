\flushleft{\textbf{Question 1:}}
Show that the matrix B'AB is symmetric or skew symmetric according as A is symmetric or skew symmetric ?\\
\textbf{Sol:} \\
\textbf{(a)when A is symmetric.}\\

 As A is symmetric matrix
		\begin{equation}
		A'=A
		\end{equation}
		Taking (B'AB)'\\
		Let
		$$AB=P$$\\
		Therefore
		$$(B'AB)'=(B'P)'$$\\
		$$(B'AB)'=P'(B')'$$\\ 		%[(B')'=B]
		$$(B'AB)'=P'B$$\\
		Putting P=AB
		$$(B'AB)'=(AB)'B$$\\
		$$(B'AB)'=B'A'(B)$$\\		%(A'=A) from eq. 1
\begin{equation}
(B'AB)'=B'AB
\end{equation}	\\
From Eq.2, it is clear that B'AB is a Symmetric matrix.\\\\

	

\textbf{(b)when A is skew symmetric.}\\

 As A is Skew symmetric matrix
		\begin{equation}
		A'=-A
		\end{equation}
		Taking (B'AB)'\\
		Let
		$$AB=Q$$\\
		Therefore
		$$(B'AB)'=(B'Q)'$$\\
		$$(B'AB)'=Q'(B')'$$\\
		$$(B'AB)'=Q'B$$\\
		Putting Q=AB
		$$(B'AB)'=(AB)'B$$\\
		$$(B'AB)'=B'A'(B)$$\\		%(A'=-A) from eq. 3
		$$(B'AB)'=B'(-A)B$$\\
\begin{equation}
(B'AB)'=-(B'AB)
\end{equation}	
From Eq.4, it is clear that B'AB is a Skew Symmetric matrix.\\
